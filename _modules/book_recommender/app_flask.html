[docs]@auth.verify_password
def verify_password(username: Any, password: Any):
    '''This function is verifing the password


    Args:
        username: Input the username
        password: Input the password

    Returns:
        username
    
    '''
    if username in users and \
            check_password_hash(users.get(username), password):
        return username
